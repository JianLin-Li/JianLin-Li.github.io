277	/**
278	 * 检测文件编码
279	 */
280	private function detectEncoding($filePath)
281	{
282		$content = file_get_contents($filePath);
283
284		// 尝试检测编码
285		$encodings = ['UTF-8', 'ISO-8859-1', 'GBK', 'GB2312', 'BIG5'];
286
287		foreach ($encodings as $encoding) {
288			if (mb_check_encoding($content, $encoding)) {
289				return $encoding;
290			}
291		}
292
293		return 'UTF-8'; // 默认返回UTF-8
294	}

296	/**
297	 * 修复单引号内的乱码内容
298	 */
299	private function fixQuotedContent($content)
300	{
301		// 使用正则表达式匹配单引号内的内容
302		$pattern = "/'([^']*)'/";
303
304		return preg_replace_callback($pattern, function ($matches) {
305			$quotedContent = $matches[1];
306			$fixedContent = $quotedContent;
307
308			// 只在单引号内进行替换
309			foreach ($this->encodingFixes as $wrong => $correct) {
310				// 确保完全匹配整个乱码字符串
311				if ($fixedContent === $wrong) {
312					$fixedContent = $correct;
313					break; // 一旦匹配就停止，避免重复替换
314				}
315			}
316
317			// 如果内容有变化，返回修复后的内容
318			if ($fixedContent !== $quotedContent) {
319				return "'" . $fixedContent . "'";
320			}
321
322			// 如果没有变化，返回原内容
323			return $matches[0];
324		}, $content);
325	}

327	/**
328	 * 修复单个SQL文件
329	 */
330	public function fixSqlFile($filePath)
331	{
332		echo "处理文件: $filePath\n";
333
334		// 备份原文件
335		$backupPath = $filePath . '.backup';
336		if (!copy($filePath, $backupPath)) {
337			echo "  错误: 无法创建备份文件\n";
338			return false;
339		}
340
341		// 检测文件编码
342		$encoding = $this->detectEncoding($filePath);
343		echo "  检测编码: $encoding\n";
344
345		// 读取文件内容
346		$content = file_get_contents($filePath);
347
348		// 如果检测到的编码不是UTF-8，尝试转换
349		if ($encoding !== 'UTF-8') {
350			$content = mb_convert_encoding($content, 'UTF-8', $encoding);
351		}
352
353		// 修复单引号内的乱码
354		$originalContent = $content;
355		$content = $this->fixQuotedContent($content);
356
357		// 检查是否有变化
358		if ($content !== $originalContent) {
359			// 统计修复的数量
360			$fixCount = 0;
361			foreach ($this->encodingFixes as $wrong => $correct) {
362				// 统计单引号内的替换次数
363				$pattern = "/'$wrong'/";
364				preg_match_all($pattern, $originalContent, $matches);
365				$fixCount += count($matches[0]);
366			}
367
368			// 保存修复后的文件
369			if (file_put_contents($filePath, $content) !== false) {
370				echo "  成功修复 {$fixCount} 处乱码，原文件已备份为: $backupPath\n";
371				return true;
372			} else {
373				echo "  错误: 无法保存修复后的文件\n";
374				// 恢复备份
375				copy($backupPath, $filePath);
376				return false;
377			}
378		} else {
379			echo "  无需修复\n";
380			// 删除备份文件（如果没有修改）
381			unlink($backupPath);
382			return true;
383		}
384	}

386	/**
387	 * 批量修复目录中的所有SQL文件
388	 */
389	public function batchFix($directory = '.')
390	{
391		echo "开始批量修复SQL文件...\n\n";
392
393		$sqlFiles = [];
394
395		// 递归查找所有SQL文件
396		$iterator = new RecursiveIteratorIterator(
397			new RecursiveDirectoryIterator($directory, RecursiveDirectoryIterator::SKIP_DOTS),
398			RecursiveIteratorIterator::SELF_FIRST
399		);
400
401		foreach ($iterator as $file) {
402			if ($file->isFile() && $file->getExtension() === 'sql') {
403				$sqlFiles[] = $file->getPathname();
404			}
405		}
406
407		if (empty($sqlFiles)) {
408			echo "未找到SQL文件\n";
409			return;
410		}
411
412		echo "找到 " . count($sqlFiles) . " 个SQL文件\n\n";
413
414		$successCount = 0;
415		$failCount = 0;
416
417		foreach ($sqlFiles as $sqlFile) {
418			if ($this->fixSqlFile($sqlFile)) {
419				$successCount++;
420			} else {
421				$failCount++;
422			}
423			echo str_repeat("-", 60) . "\n";
424		}
425
426		echo "\n修复完成!\n";
427		echo "成功: $successCount 个文件\n";
428		echo "失败: $failCount 个文件\n";
429	}
